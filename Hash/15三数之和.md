不用哈希法，用双指针：
```go
func threeSum(nums []int) [][]int {
    // 将数组排序
    sort.Ints(nums)
    res := [][]int{}
    for i := 0; i < len(nums) - 2; i++ {
        n1 := nums[i]
        // 如果最小的数字已经大于0，则无可能有符合条件的数组
        if n1 > 0 { break }
        // 跳过重复的数字
        if i > 0 && nums[i - 1] == n1 { continue }
        // 双指针比较剩下的两个数
        l, r := i + 1, len(nums) - 1
        for l < r {
            n2, n3 := nums[l], nums[r]
            sum := n1 + n2 + n3
            if sum == 0 {
                res = append(res, []int{n1, n2, n3})
                // 同样跳过重复的数字
                for l < r && nums[l] == nums[l+1] { l++ }
                for l < r && nums[r] == nums[r-1] { r-- }
                l++
                r--
            } else if sum < 0 {
                l++
            } else {
                r--
            }
        }
    }
    return res
}
```
