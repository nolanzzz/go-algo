func minCostClimbingStairs(cost []int) int {
	n := len(cost)
	dp := make([]int, n) // dp[i]为达到第i阶时的最小花费
	dp[0] = cost[0]
	dp[1] = cost[1]
	for i := 2; i < n; i++ {
		dp[i] = min(dp[i-2], dp[i-1]) + cost[i]
	}
	return min(dp[n-1], dp[n-2])
}

func min(x, y int) int {
	if x < y {
		return x
	}
	return y
}
